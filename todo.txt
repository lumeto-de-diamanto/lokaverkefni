# Graph stuff

def add_vertex(g, v):
  if v in g:
    raise Exception("Hnúturinn er þegar í netinu")
  else:
    g[v] = []
  return g

def add_edge(g, fro, to, data):
  g[fro].append([data, to])
  return g
